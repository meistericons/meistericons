# @meistericons/vue-latest

## Installtion

### CDN

```bash
<link rel="https://unpkg.com/@meistericons/vue-latest@latest">
```

### npm

```npm
npm install @meistericons/vue-latest --save-dev
```

## Usage

```vue
import {Airpods} from '@meistericons/vue-latest'

export default function App() {
    return <Airpods/>
}
```
