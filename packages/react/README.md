# @meistericons/react

## Installtion

### CDN

```bash
<link rel="https://unpkg.com/@meistericons/react@latest">
```

### npm

```npm
npm install @meistericons/react --save-dev
```

## Usage

```jsx
import {Airpods} from '@meistericons/react'

export default function App() {
    return <Airpods/>
}
```
